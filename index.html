    <script>

        async function fetchLeaderboard() {

            try {

                // Fetch the JSON data from Google Sheets

                const response = await fetch('https://docs.google.com/spreadsheets/d/2PACX-1vSDPcUO91Ef4TzwRtP-vYLU1_HHpjxhkolrAcieuH6DNBwmcfVQ0CjouaB0IlG7vqqhCs-RLFbTCMuL/gviz/tq?tqx=out:json');

                const data = await response.text();


                // Parse the JSON response from Google Sheets

                const jsonData = JSON.parse(data.substr(47).slice(0, -2));

                

                // Extract leaderboard data

                const leaderboard = jsonData.table.rows.map(row => {

                    return {

                        rank: row.c[0]?.v,

                        username: row.c[1]?.v,

                        points: row.c[2]?.v

                    };

                });


                // Populate the table with leaderboard data

                const tbody = document.querySelector('tbody');

                tbody.innerHTML = ''; // Clear any existing rows

                leaderboard.forEach(entry => {

                    const tr = document.createElement('tr');

                    tr.innerHTML = `<td>${entry.rank}</td><td>${entry.username}</td><td>${entry.points}</td>`;

                    tbody.appendChild(tr);

                });

            } catch (error) {

                console.error("Error fetching leaderboard data:", error);

            }

        }


        // Call the function to load the leaderboard data

        fetchLeaderboard();

    </script>
