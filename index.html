        async function fetchLeaderboardData() {

            const response = await fetch('https://docs.google.com/spreadsheets/d/e/2PACX-1vSDPcUO91Ef4TzwRtP-vYLU1_HHpjxhkolrAcieuH6DNBwmcfVQ0CjouaB0IlG7vqqhCs-RLFbTCMuL/gviz/tq?tqx=out:json’); // Replace with your Google Sheets JSON URL

            const data = await response.text();

            const jsonData = JSON.parse(data.substr(47).slice(0, -2));

            const leaderboard = jsonData.table.rows;


            const leaderboardTable = document.getElementById('leaderboardTable').getElementsByTagName('tbody')[0];

            leaderboardTable.innerHTML = ''; // Clear existing table data


            leaderboard.forEach((item, index) => {

                const row = leaderboardTable.insertRow();

                row.insertCell(0).innerText = index + 1; // Rank

                row.insertCell(1).innerText = item.c[0].v; // User

                row.insertCell(2).innerText = item.c[1].v; // Plays

            });

        }


        // Fetch leaderboard data every 10 minutes

        setInterval(fetchLeaderboardData, 600000);

        fetchLeaderboardData(); // Initial fetch

    </script>
